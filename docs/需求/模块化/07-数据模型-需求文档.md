```csharp
public static class GatePaths {

    public static string PluginsDir    => Path.Combine(DataDir, "plugins");
    public static string BackupsDir    => Path.Combine(DataDir, "backups"); // P1
}
```

## 2. GlobalConfig（config.json）

```json
{
  "autoSaveHistory": true,
  "maxHistoryCount": 20,
  "defaultTestUrl": "https://www.google.com",
  "defaultTimeout": 10000,
  "defaultPreset": "",
  "pluginRegistryUrl": "",
  "colorEnabled": true,
  "proxyExpiresAt": null
}
```

```csharp
public class GlobalConfig {
    public bool   AutoSaveHistory    { get; set; } = true;
    public int    MaxHistoryCount    { get; set; } = 20;
    public string DefaultTestUrl     { get; set; } = "https://www.google.com";
    public int    DefaultTimeout     { get; set; } = 10000;
    public string DefaultPreset      { get; set; } = "";
    public string PluginRegistryUrl  { get; set; } = "";
    public bool   ColorEnabled       { get; set; } = true;
    public DateTime? ProxyExpiresAt  { get; set; } = null; // P1
}
```

**读写规则**：
- 文件不存在时使用默认值，第一次写入时创建
- 文件损坏（非法 JSON）时使用默认值并输出 WARN，不抛异常
- 使用 `System.Text.Json` 序列化，`PropertyNamingPolicy.CamelCase`
- `ProxyExpiresAt`：**仅在 `gate` 无参启动或 `gate info` 时检查**，其他命令执行前不检查（A13：方案A）

## 3. Profile（profiles/{name}.json）

```json
{
  "name": "office",
  "description": "公司代理",
  "isDefault": true,
  "createdAt": "2026-03-18T09:00:00Z",
  "updatedAt": "2026-03-18T10:00:00Z",
  "globalProxy": {
    "httpProxy": "http://proxy.company.com:8080",
    "httpsProxy": "http://proxy.company.com:8080",
    "noProxy": "localhost,127.0.0.1"
  },
  "toolProxies": [
    { "toolName": "git", "proxy": "http://proxy.company.com:8080" }
  ]
}
```

```csharp
public class Profile {
    public string   Name        { get; set; } = "";
    public string   Description { get; set; } = "";
    public bool     IsDefault   { get; set; } = false;
    public DateTime CreatedAt   { get; set; }
    public DateTime UpdatedAt   { get; set; }
    public ProxyConfig GlobalProxy { get; set; } = new();
    public List<ToolProxyEntry> ToolProxies { get; set; } = new();
}
public record ProxyConfig(string? HttpProxy, string? HttpsProxy, string? NoProxy);
public record ToolProxyEntry(string ToolName, string Proxy);
```

**文件命名**：`{name}.json`，`name` 仅含字母/数字/连字符/下划线。

## 4. HistoryEntry（history.json）

```json
[
  { "proxy": "http://127.0.0.1:7890", "usedAt": "2026-03-18T10:00:00Z" }
]
```

```csharp
public record HistoryEntry(string Proxy, DateTime UsedAt);
```

**规则**：最多 `MaxHistoryCount` 条；相同 proxy 值的重复条目合并（保留最新 usedAt）；文件损坏视为空历史，覆盖重写。

## 5. ToolDescriptor（EmbeddedResource tools/**/*.json）

```json
{
  "name": "npm",
  "displayName": "npm",
  "category": "包管理器",
  "description": "Node.js 包管理器",
  "config": {
    "format": "keyvalue",
    "configFiles": [
      { "path": "~/.npmrc", "platform": "all" }
    ],
    "proxyKey": "proxy",
    "httpsProxyKey": "https-proxy",
    "removeOnClear": true
  },
  "probePaths": [
    { "path": "~/.npmrc", "platform": "all" }
  ]
}
```

```csharp
public class ToolDescriptor {
    public string Name        { get; set; } = "";
    public string DisplayName { get; set; } = "";
    public string Category    { get; set; } = "";
    public string Description { get; set; } = "";
    public ToolConfig Config  { get; set; } = new();
    public List<ProbePath> ProbePaths { get; set; } = new();
}
public class ToolConfig {
    public string Format        { get; set; } = ""; // keyvalue|ini|json|toml
    public List<ConfigFile> ConfigFiles { get; set; } = new();
    public string? ProxyKey     { get; set; }
    public string? HttpsProxyKey { get; set; }
    public string? Section      { get; set; }  // ini 格式的 section 名
    public string? JsonPath     { get; set; }  // json 格式的 JPath
    public bool   RemoveOnClear { get; set; } = true;
}
public record ConfigFile(string Path, string Platform); // platform: all|windows|linux|macos
public record ProbePath(string Path, string Platform);
```

## 9. 序列化规范

| 规范项 | 值 |
|--------|----|
| 库 | `System.Text.Json` |
| 命名策略 | `JsonNamingPolicy.CamelCase` |
| 枚举序列化 | 字符串（`JsonStringEnumConverter`） |
| 日期格式 | ISO 8601（`2026-03-18T10:00:00Z`，UTC） |
| null 值 | 默认序列化（保留 null 字段） |
| 格式化 | `WriteIndented = true`（所有持久化文件） |
| 文件编码 | UTF-8 无 BOM |
| 写入方式 | 先写 `.tmp`，再 `File.Replace`（原子操作） |

## 10. 验收标准

- [ ] `config.json` 损坏时 Gate 使用默认值启动，不崩溃
- [ ] `profile.json` 所有字段均有默认值，缺字段时反序列化不报错
- [ ] `history.json` 超出 MaxHistoryCount 时自动截断最旧条目
- [ ] 所有文件写入使用原子操作（`.tmp` → rename）
- [ ] `ToolDescriptor` 的 `platform` 字段正确过滤当前平台的 configFiles
